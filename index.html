import face_recognition import cv2 import os from google.colab.patches import cv2_imshow #Importamos cv2_imshow # Cargar imágenes y codificar rostros known_face_encodings = [] known_face_names = [] # Obtener la ruta absoluta del directorio actual current_directory = os.path.dirname(os.path.abspath(__file__)) # Función para cargar imágenes y codificar rostros def load_images_and_encode(directory): for filename in os.listdir(directory): if filename.endswith((".jpg", ".jpeg", ".png")): image_path = os.path.join(directory, filename) image = face_recognition.load_image_file(image_path) face_encoding = face_recognition.face_encodings(image) if len(face_encoding) > 0: known_face_encodings.append(face_encoding[0]) known_face_names.append(os.path.splitext(filename)[0]) else: print(f"Advertencia: No se detectó ningún rostro en {filename}") # Cargar imágenes desde el directorio "imagenes" imagenes_directory = os.path.join(current_directory, "imagenes") load_images_and_encode(imagenes_directory) # Inicializar la cámara video_capture = cv2.VideoCapture(0) if not video_capture.isOpened(): print("Error: No se pudo abrir la cámara.") exit() # Bucle principal while True: ret, frame = video_capture.read() if not ret: print("Error: No se pudo obtener el frame.") break # Redimensionar el frame para mejorar el rendimiento small_frame = cv2.resize(frame, (0, 0), fx=0.25, fy=0.25) rgb_small_frame = cv2.cvtColor(small_frame, cv2.COLOR_BGR2RGB) # Encontrar rostros y codificarlos face_locations = face_recognition.face_locations(rgb_small_frame) face_encodings = face_recognition.face_encodings(rgb_small_frame, face_locations) face_names = [] for face_encoding in face_encodings: # Comparar rostros matches = face_recognition.compare_faces(known_face_encodings, face_encoding) name = "Desconocido" # Encontrar la mejor coincidencia if True in matches: first_match_index = matches.index(True) name = known_face_names[first_match_index] face_names.append(name) # Mostrar resultados for (top, right, bottom, left), name in zip(face_locations, face_names): top *= 4 right *= 4 bottom *= 4 left *= 4 cv2.rectangle(frame, (left, top), (right, bottom), (0, 0, 255), 2) cv2.putText(frame, name, (left + 6, bottom - 6), cv2.FONT_HERSHEY_DUPLEX, 0.8, (255, 255, 255), 1) # Mostrar el frame resultante cv2_imshow(frame) # Utilizamos cv2_imshow en vez de cv2.imshow # Salir del bucle si se presiona 'q' if cv2.waitKey(1) & 0xFF == ord('q'): break # Liberar recursos video_capture.release() cv2.destroyAllWindows() # Liberar recursos video_capture.release() cv2.destroyAllWindows()
